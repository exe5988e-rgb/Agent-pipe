def read_current():
    """Read current ports from existing files."""
    a_ip, a_port = "192.168.29.48", "37891"
    b_ip, b_port = "192.168.29.110", "39991"

    bridge = os.path.join(HOME, "claude_bridge.py")
    if os.path.exists(bridge):
        m = re.search(r'SERIAL\s*=\s*"([^":]+):(\d+)"', open(bridge).read())
        if m:
            b_ip, b_port = m.group(1), m.group(2)

    for llm_path in [
        "test/jarvis-agent/reasoning/llm_client.py",
        "workflows/test/jarvis-agent/reasoning/llm_client.py",
    ]:
        full = os.path.join(HOME, llm_path)
        if os.path.exists(full):
            text = open(full).read()
            m = re.search(r'PHONE_A_SERIAL\s*=\s*"([^":]+):(\d+)"', text)
            if m:
                a_ip, a_port = m.group(1), m.group(2)
            m2 = re.search(r'PHONE_B_URL\s*=\s*"http://([^/:]+):(\d+)"', text)
            if m2:
                b_ip, b_port = m2.group(1), m2.group(2)
            break

    return a_ip, a_port, b_ip, b_port

def ask_device(name, color, cur_ip, cur_port):
    print(f"  {color}● {name}{RST}")
    print(f"  {DIM}Current: {cur_ip}:{cur_port}{RST}\n")

    ip_in = input(f"  IP   [{cur_ip}]: ").strip()
    ip = ip_in if ip_in else cur_ip

    port_in = input(f"  Port [{cur_port}]: ").strip()
    if port_in and re.match(r'^\d+$', port_in):
        port = port_in
    else:
        port = cur_port
        if port_in:
            print(f"  {Y}Invalid port — keeping {cur_port}{RST}")

    print()
    return ip, port

def apply_simple(rel, label, pattern, repl, a_ip, a_port, b_ip, b_port):
    full = os.path.join(HOME, rel)
    if not os.path.exists(full):
        return

    text = open(full).read()
    r = repl.replace("{A_IP}", a_ip).replace("{A_PORT}", a_port) \
            .replace("{B_IP}", b_ip).replace("{B_PORT}", b_port)
    new = re.sub(pattern, r, text)

    if new != text:
        open(full, "w").write(new)
        print(f"  {G}✓{RST}  {label:<20} {DIM}{rel}{RST}")
    else:
        print(f"  {DIM}~  {label:<20} no match — {rel}{RST}")

def apply_init_sh(rel, a_ip, a_port, b_ip, b_port):
    """Replace all IP:port pairs in shell scripts mapping old→new."""
    full = os.path.join(HOME, rel)
    if not os.path.exists(full):
        return

    text = open(full).read()
    original = text

    # collect all unique IP:port pairs in file
    pairs = re.findall(r'(192\.168\.\d+\.\d+):(\d+)', text)
    if not pairs:
        print(f"  {DIM}~  no IP:port found — {rel}{RST}")
        return

    # heuristic: lower port → Phone A, higher port or second IP → Phone B
    # just replace all — user controls which is A vs B via prompt
    unique = list(dict.fromkeys(pairs))  # preserve order, dedupe

    if len(unique) >= 2:
        old_a_ip, old_a_port = unique[0]
        old_b_ip, old_b_port = unique[1]
        text = text.replace(f"{old_a_ip}:{old_a_port}", f"{a_ip}:{a_port}")
        text = text.replace(f"{old_b_ip}:{old_b_port}", f"{b_ip}:{b_port}")
    elif len(unique) == 1:
        old_ip, old_port = unique[0]
        text = text.replace(f"{old_ip}:{old_port}", f"{b_ip}:{b_port}")

    if text != original:
        open(full, "w").write(text)
        print(f"  {G}✓{RST}  ports updated      {DIM}{rel}{RST}")
    else:
        print(f"  {DIM}~  no changes        {rel}{RST}")

def confirm(a_ip, a_port, b_ip, b_port):
    print(LINE)
    print(f"\n  {W}Summary:{RST}\n")
    print(f"  {B}Phone A (OPPO){RST}              {a_ip}:{a_port}")
    print(f"  {C}Phone B (Galaxy A10s){RST}       {b_ip}:{b_port}")
    print()
    ans = input(f"  Apply? [{G}Y{RST}/{R}n{RST}]: ").strip().lower()
    return ans in ("", "y", "yes")

def main():
    banner()

    a_ip, a_port, b_ip, b_port = read_current()

    print(f"  {W}Current:{RST}")
    print(f"  {B}Phone A{RST}  {a_ip}:{a_port}")
    print(f"  {C}Phone B{RST}  {b_ip}:{b_port}")
    print(f"\n{LINE}\n")
    print(f"  Press {G}Enter{RST} to keep current value\n")

    print(f"{B}  PHONE A — OPPO (agent loop){RST}")
    a_ip, a_port = ask_device("Phone A", B, a_ip, a_port)

    print(LINE)
    print()

    print(f"{C}  PHONE B — Samsung Galaxy A10s (bridge){RST}")
    b_ip, b_port = ask_device("Phone B", C, b_ip, b_port)

    if not confirm(a_ip, a_port, b_ip, b_port):
        print(f"\n  {Y}Cancelled.{RST}\n")
        sys.exit(0)

    print(f"\n{LINE}")
    print(f"  {W}Applying...{RST}\n")

    for rel, patches in PATCHES:
        for label, pattern, repl in patches:
            if repl is None:
                apply_init_sh(rel, a_ip, a_port, b_ip, b_port)
                break
            else:
                apply_simple(rel, label, pattern, repl, a_ip, a_port, b_ip, b_port)

    print(f"\n{LINE}")
    print(f"\n  {G}Done.{RST}\n")
    print(f"  {DIM}Phone B — restart bridge:{RST}")
    print(f"  pkill -f claude_bridge && python3 ~/claude_bridge.py\n")

if __name__ == "__main__":
    main()

# Regex to extract app name from TASK line
# Matches: "TASK: Open WhatsApp", "TASK: Launch Chrome", "TASK: open youtube"
_OPEN_TASK_RE = re.compile(
    r"TASK\s*:\s*(?:open|launch|start|go to|goto)\s+(.+?)(?:\n|$)",
    re.IGNORECASE
)

_SCREEN_RE = re.compile(r"Screen\s*:\s*(.+)", re.IGNORECASE)

_LAUNCHER_SCREENS = {
    "home screen", "launcher", "home screen launcher",
    "app drawer", "home", "lock screen",
}

def _is_on_launcher(prompt: str) -> bool:
    """Returns True only if the prompt Screen line shows home/launcher."""
    m = _SCREEN_RE.search(prompt)
    if not m:
        return False
    screen = m.group(1).strip().lower()
    return any(s in screen for s in _LAUNCHER_SCREENS)

def extract_task_app(prompt: str):
    """
    Return package only if:
      1. Task starts with open/launch/start
      2. Current screen is home/launcher
    Otherwise return None — fall through to Claude.
    """
    if not _is_on_launcher(prompt):
        return None
    m = _OPEN_TASK_RE.search(prompt)
    if not m:
        return None
    app_name = m.group(1).strip().lower()
    first_word = app_name.split()[0] if app_name else ""
    if first_word in APP_PACKAGES:
        return APP_PACKAGES[first_word]
    for key, pkg in APP_PACKAGES.items():
        if key in app_name:
            return pkg
    return None

def adb(*args, timeout=20) -> subprocess.CompletedProcess:
    cmd = [ADB, "-s", SERIAL] + list(str(a) for a in args)
    return subprocess.run(cmd, capture_output=True, timeout=timeout)

def shell(*args, timeout=20) -> str:
    r = adb("shell", *args, timeout=timeout)
    return r.stdout.decode(errors="ignore").strip()

def ui_dump() -> str:

def paste_via_clipper(text: str):
    print(f"[bridge] clipboard paste ({len(text)} chars)...")
    encoded = base64.b64encode(text.encode("utf-8")).decode("ascii")
    write_cmd = f"echo '{encoded}' | base64 -d > {PROMPT_TMP}"
    result = adb("shell", write_cmd)
    if result.returncode != 0:
        print(f"[bridge] WARNING: file write failed: {result.stderr.decode()}")
    time.sleep(0.15)
    clip_cmd = f"am broadcast -a clipper.set --es text \"$(cat {PROMPT_TMP})\""
    result = adb("shell", clip_cmd)
    out = result.stdout.decode(errors="ignore")
    print(f"[bridge] clipper set: {out.strip()[:80]}")
    time.sleep(0.3)
    shell("input", "keyevent", "279")
    time.sleep(0.2)
    print("[bridge] paste done")

def adb_broadcast(text: str):

def _sanitize_wordbyword(text: str) -> str:
    return (text
            .replace("'", "")
            .replace('"', "")
            .replace("\\", "")
            .replace("`", "")
            .replace("&", "and")
            .replace("|", " ")
            .replace(";", " ")

            time.sleep(0.1)

# ── EditText detection ────────────────────────────────────────────────────────

def find_edittext(xml: str):
    patterns = [
        r'class="android\.widget\.EditText"[^>]*bounds="\[(\d+),(\d+)\]\[(\d+),(\d+)\]"',
        r'class="android\.widget\.MultiAutoCompleteTextView"[^>]*bounds="\[(\d+),(\d+)\]\[(\d+),(\d+)\]"',
        r'class="com\.facebook\.react\.views\.textinput\.ReactEditText"[^>]*bounds="\[(\d+),(\d+)\]\[(\d+),(\d+)\]"',
        r'(?:hint-text="[^"]*[Mm]essage[^"]*"|content-desc="[^"]*[Mm]essage[^"]*")[^>]*bounds="\[(\d+),(\d+)\]\[(\d+),(\d+)\]"',
    ]
    for pat in patterns:
        m = re.search(pat, xml)
        if m:
            x1, y1, x2, y2 = map(int, m.groups())
            cx, cy = (x1 + x2) // 2, (y1 + y2) // 2
            print(f"[bridge] EditText found via pattern: class match, center=({cx},{cy})")
            return cx, cy
    for m in re.finditer(
        r'focusable="true"[^>]*bounds="\[(\d+),(\d+)\]\[(\d+),(\d+)\]"', xml
    ):
        x1, y1, x2, y2 = map(int, m.groups())
        cy = (y1 + y2) // 2
        if cy > SCREEN_H * 0.70:
            cx = (x1 + x2) // 2
            print(f"[bridge] EditText found via focusable fallback, center=({cx},{cy})")
            return cx, cy
    return None

def _claude_is_foreground() -> bool:
    win_out = shell("dumpsys", "window", "windows")

    return False

def is_still_generating(dump_xml: str) -> bool:
    return 'content-desc="Stop"' in dump_xml or "Stop generating" in dump_xml

def extract_last_response(dump_xml: str) -> str:
    texts = re.findall(r'text="([^"]{3,})"', dump_xml)
    candidates = [
        t for t in texts
        if t not in SKIP_TEXTS
        and not any(t.startswith(p) for p in SKIP_PREFIXES)
        and not t.startswith("  ")
    ]
    return candidates[-1] if candidates else ""

def extract_action_from_response(full_text: str) -> str:
    lines = full_text.strip().splitlines()
    action_re = re.compile(
        r"^\s*(tap\s+\d+|tap\s+\d+[,x]\d+|type\s+.+|swipe\s+(up|down|left|right)"
        r"|back|home|enter|done)\s*$",
        re.I
    )
    for line in lines:
        if action_re.match(line.strip()):
            return line.strip()
    for line in reversed(lines):
        if line.strip():
            return line.strip()
    return full_text.strip()

def wait_for_response() -> str:
    print("[bridge] Waiting for response...")
    last_text = ""
    stable    = 0
    deadline  = time.time() + MAX_WAIT
    while time.time() < deadline:
        dump = ui_dump()
        if is_still_generating(dump):
            print("[bridge] ...generating...")
            stable = 0
            time.sleep(POLL_INTERVAL)
            continue
        current = extract_last_response(dump)
        if len(current) > len(last_text):
            last_text = current
            stable    = 0
            print(f"[bridge] Growing: {current[:60]!r}")
        elif current and current == last_text:
            stable += 1
            print(f"[bridge] Stable {stable}/{STABLE_COUNT}: {current[:60]!r}")
            if stable >= STABLE_COUNT:
                action = extract_action_from_response(current)
                print(f"[bridge] Done. Action: {action!r}")
                return action
        time.sleep(POLL_INTERVAL)
    print("[bridge] Timeout")
    return last_text or "home"

class BridgeHandler(BaseHTTPRequestHandler):

    def log_message(self, fmt, *args):
        print(f"[bridge] {fmt % args}")

    def send_json(self, code, data):
        body = json.dumps(data).encode()
        self.send_response(code)
        self.send_header("Content-Type", "application/json")
        self.send_header("Access-Control-Allow-Origin", "*")
        self.end_headers()
        self.wfile.write(body)

    def do_GET(self):
        if self.path == "/health":
            self.send_json(200, {"success": True, "service": "claude_bridge",
                                 "screen": f"{SCREEN_W}x{SCREEN_H}",
                                 "serial": SERIAL,
                                 "opt4_apps": list(APP_PACKAGES.keys())})
        else:
            self.send_json(404, {"error": "not found"})

    def do_POST(self):
        length = int(self.headers.get("Content-Length", 0))
        body   = self.rfile.read(length)
        try:
            data = json.loads(body.decode())
        except Exception:
            self.send_json(400, {"success": False, "error": "invalid JSON"})
            return

        if self.path == "/chat":
            prompt = data.get("prompt", "").strip()
            if not prompt:
                self.send_json(400, {"success": False, "error": "missing prompt"})
                return

            print(f"\n[bridge] ══ PROMPT ══\n{prompt[:300]}\n{'='*40}")
            try:
                result = handle_prompt(prompt)
                print(f"[bridge] ══ ACTION ══\n{result}\n{'='*40}\n")
                self.send_json(200, {"success": True, **result})
            except Exception as e:
                import traceback
                traceback.print_exc()
                self.send_json(500, {"success": False, "error": str(e)})
        else:
            self.send_json(404, {"error": "not found"})

import os
import re
import sys
import subprocess

# ===== Only match TERMUXFILE headers =====
# Example: //===== TERMUXFILE: path =====
FILE_HEADER = re.compile(
    r"^[ \t]*//\s*={5,}\s*TERMUXFILE:\s*(.+?)\s*={5,}\s*$",
    re.IGNORECASE | re.MULTILINE
)

def sanitize(path):
    path = path.strip().replace("\\", "/")

    if ".." in path or path.startswith("/") or re.match(r"^[A-Za-z]:", path):
        raise ValueError(f"Unsafe path: {path}")

    return path

def parse_ai_output(text):
    matches = list(FILE_HEADER.finditer(text))
    files = {}

    for i, match in enumerate(matches):
        file_path = sanitize(match.group(1))

        start = match.end()
        end = matches[i + 1].start() if i + 1 < len(matches) else len(text)

        code = text[start:end].strip("\n")

        if not code.strip():
            continue

        if file_path in files:
            print(f"🔁 Overwriting duplicate file: {file_path}")

        files[file_path] = code

    return files

def write_files(files):
    for path, content in files.items():
        dir_path = os.path.dirname(path)
        if dir_path:
            os.makedirs(dir_path, exist_ok=True)

        with open(path, "w", encoding="utf-8", newline="\n") as f:
            f.write(content.rstrip() + "\n")

        print(f"✅ Written: {path}")

def is_git_repo():
    try:
        subprocess.run(
            ["git", "rev-parse", "--is-inside-work-tree"],
            stdout=subprocess.DEVNULL,
            stderr=subprocess.DEVNULL,
            check=True
        )
        return True
    except subprocess.CalledProcessError:
        return False

def auto_commit(message="AI: generate project files"):
    if not is_git_repo():
        print("ℹ️ Not a git repository, skipping commit")
        return

    try:
        subprocess.run(["git", "add", "."], check=True)
        subprocess.run(["git", "commit", "-m", message], check=True)
        print("📝 Git commit created")
    except subprocess.CalledProcessError:
        print("ℹ️ Nothing to commit")

def main():
    if len(sys.argv) != 2:
        print("Usage:")
        print("  python ai-code.py ai-output.txt")
        sys.exit(1)

    input_file = sys.argv[1]

    if not os.path.exists(input_file):
        print(f"❌ File not found: {input_file}")
        sys.exit(1)

    with open(input_file, "r", encoding="utf-8") as f:
        text = f.read()

    files = parse_ai_output(text)

    print(f"\n📦 Detected {len(files)} TERMUXFILE entries")

    for path in files:
        print(" -", path)

    if not files:
        print("⚠️ No TERMUXFILE entries detected")
        sys.exit(1)

    write_files(files)
    auto_commit()

if __name__ == "__main__":
    main()

#!/usr/bin/env python3

import time
import json
import subprocess
from pathlib import Path

WATCH_DIR = Path.home()
CONFIG = Path.home() / "repo_target.json"

OUTPUT = "repo source dev b.txt"
MAX_SIZE = 500 * 1024
INTERVAL = 2


def load_repo():
    return json.loads(CONFIG.read_text())


def file_state(root):

    state = {}

    for f in root.rglob("*"):
        if f.is_file():
            state[str(f)] = f.stat().st_mtime

    return state


def generate_dump(root, output):

    with open(output, "w", encoding="utf-8") as out:

        for f in root.rglob("*"):

            if not f.is_file():
                continue

            if ".git" in str(f):
                continue

            try:

                rel = f.relative_to(root)

                out.write(f"\n//===== TERMUXFILE: {rel} =====\n")
                out.write(f.read_text(errors="ignore"))
                out.write("\n")

            except:
                pass


def split_dump(repo_path):

    dump_file = repo_path / OUTPUT

    if not dump_file.exists():
        return

    text = dump_file.read_text(errors="ignore")

    sections = text.split("//===== TERMUXFILE:")

    # remove old chunks
    for old in repo_path.glob("repo source dev b_*.txt"):
        old.unlink()

    part = 1
    current = ""

    for section in sections:

        if not section.strip():
            continue

        block = "//===== TERMUXFILE:" + section

        if len((current + block).encode()) > MAX_SIZE:

            outfile = repo_path / f"repo source dev b_{part:03}.txt"
            outfile.write_text(current)

            print("Created:", outfile.name)

            part += 1
            current = block

        else:

            current += block

    if current:

        outfile = repo_path / f"repo source dev b_{part:03}.txt"
        outfile.write_text(current)

        print("Created:", outfile.name)


def git_push(repo):

    path = repo["path"]

    subprocess.run(["git", "-C", path, "add", "."], check=False)
    subprocess.run(["git", "-C", path, "commit", "-m", "auto repo dump"], check=False)
    # --set-upstream handles first push and all subsequent pushes
    subprocess.run(["git", "-C", path, "push", "--set-upstream", "origin", "main"], check=False)


def main():

    print("Starting repo dump watcher\n")

    repo = load_repo()

    print("Current repo:", repo["name"])

    last = file_state(WATCH_DIR)

    while True:

        time.sleep(INTERVAL)

        new = file_state(WATCH_DIR)

        if new != last:

            repo = load_repo()

            print("Change detected")
            print("Target repo:", repo["name"])

            repo_path = Path(repo["path"])

            dump_file = repo_path / OUTPUT

            generate_dump(WATCH_DIR, dump_file)

            split_dump(repo_path)

            git_push(repo)

            print("Push complete\n")

            last = new


if __name__ == "__main__":
    main()
